1.  Introduction

   NETCONF [NETCONF] configuraion protocol defines Extensible Markup
   Language [XML] based mechanisms to install, manipulate, and delete
   the configuration of network devices.  NetConf protocol can be
   conceptually partitioned into four layers:

                Layer                      Example
            +-------------+      +-----------------------------+
            |   Content   |      |     Configuration data      |
            +-------------+      +-----------------------------+
                   |                           |
            +-------------+      +-----------------------------+
            | Operations  |      | <get-config>, <edit-config> |
            +-------------+      +-----------------------------+
                   |                           |
            +-------------+      +-----------------------------+
            |     RPC     |      |    <rpc>, <rpc-reply>       |
            +-------------+      +-----------------------------+
                   |                           |
            +-------------+      +-----------------------------+
            | Application |      |   BEEP, SSH, SSL, console   |
            |   Protocol  |      |                             |
            +-------------+      +-----------------------------+

   Framework for Netconf Data Models [NETMOD] provides guidance for the
   creation of Netconf data model.  This document defines initial set of
   core data types that will be useful in defining NetConf content.

2.  NetConf Core Data Types

   The Netconf architecture supports a clear separation between content
   and protocol.  There are major advantages to ensuring that every data
   model writer do not have to invent new data types.

   The following are initial set of reusable data types.



           <simpleType name="MD5">
                   <annotation>
                           <documentation>
                                   MD5 Message-Digest Algorithm
                             </documentation>
                   </annotation>
                   <restriction base="string">
                           <pattern value="[0-9a-zA-Z]{32}"/>
                   </restriction>
           </simpleType>
           <simpleType name="IPAddress">
                   <annotation>
                           <documentation>
                                   IP Address
                             </documentation>
                   </annotation>
                   <restriction base="string">
                           <maxLength value="39"/>
                   </restriction>
           </simpleType>
           <simpleType name="IPV4Address">
                   <annotation>
                           <documentation>
                                   IPV4 Address
                             </documentation>
                   </annotation>
                   <restriction base="dt:IPAddress">
                           <minLength value="7"/>
                           <maxLength value="15"/>
                           <pattern value=
                           "(([0-1]?[0-9]?[0-9]|2?[0-5]?[0-5])\.){2}
                  ([0-1]?[0-9]?[0-9]|2?[0-5]?[0-5])"/>
                   </restriction>
           </simpleType>
           <simpleType name="IPV6AddressFull">
                   <restriction base="dt:IPAddress">
                           <pattern value=
                   "(([0-9a-fA-F]{1,4}:){7})([0-9a-fA-F]{1,4})"/>

                   </restriction>
           </simpleType>
           <simpleType name="IPV6AddressMixed">
                   <restriction base="dt:IPAddress">
                           <pattern value="(([0-9a-fA-F]{1,4}:){6})
               (([0-9a-fA-F]{1,4}:[0-9a-fA-F]{1,4})|
               ([0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}))"/>
                   </restriction>
           </simpleType>
           <simpleType name="IPV6AddressShortened">
                   <restriction base="dt:IPAddress">
                           <pattern value=
                   "(([0-9a-fA-F]{1,4}:)*|([0-9a-fA-F]{1,4}))*(::)
            (([0-9a-fA-F]{1,4}:)*|([0-9a-fA-F]{1,4}))*"/>
                   </restriction>
           </simpleType>
           <simpleType name="IPV6Address">
                   <annotation>
                           <documentation>
                                   IPV6 Address RFC 3513
                             </documentation>
                   </annotation>
                   <union memberTypes="dt:IPV6AddressFull
                                   dt:IPV6AddressMixed
                                   dt:IPV6AddressShortened"/>
           </simpleType>
           <simpleType name="DomainName">
                   <annotation>
                           <documentation>
                                   Domain Name RFC 1034
                             </documentation>
                   </annotation>
                   <restriction base="string">
                           <pattern value=
                           "([\p{L}\p{N}]*\.)*[\p{L}\p{N}]?"/>
                   </restriction>
           </simpleType>
           <simpleType name="EthernetAddress">
                   <annotation>
                           <documentation>
                                   Ethernet Address, IEEE 802.3
                             </documentation>
                   </annotation>
                   <restriction base="string">
                           <pattern value="[0-9a-fA-F]{12}"/>
                   </restriction>
           </simpleType>
           <simpleType name="PORT">

                   <annotation>
                           <documentation>
                                   Port, RFC 793, RFC 768
                             </documentation>
                   </annotation>
                   <restriction base="unsignedInt">
                           <minInclusive value="1"/>
                           <maxInclusive value="65535"/>
                   </restriction>
           </simpleType>
           <simpleType name="AdminStatus">
                   <annotation>
                           <documentation>
                                   Admin Status
                             </documentation>
                   </annotation>
                   <restriction base="string">
                           <enumeration value="Up"/>
                           <enumeration value="Down"/>
                           <enumeration value="Testing"/>
                   </restriction>
           </simpleType>
           <simpleType name="OperStatus">
                   <annotation>
                           <documentation>
                                   Operational Status
                             </documentation>
                   </annotation>
                   <restriction base="string">
                           <enumeration value="Up"/>
                           <enumeration value="Down"/>
                           <enumeration value="Testing"/>
                           <enumeration value="Unknown"/>
                           <enumeration value="Dormant"/>
                           <enumeration value="NotPresent"/>
                           <enumeration value="LowerLayerDown"/>
                   </restriction>
           </simpleType>
           <simpleType name="E164CountryCode">
                   <annotation>
                           <documentation>
                                   ITU-T E.164 Country Code
                             </documentation>
                   </annotation>
                   <restriction base="string">
                           <pattern value="[0-9]{3}"/>
                   </restriction>
           </simpleType>
